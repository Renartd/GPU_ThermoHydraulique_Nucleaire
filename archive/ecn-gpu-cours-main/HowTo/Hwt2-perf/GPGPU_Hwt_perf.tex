 % --------- gprof ---------
\begin{frame}[plain,fragile]
\frametitle{Profilage sur CPU : GProf}

	\begin{enumerate}
		\item  Compiler le code avec l'option {\tt -pg}
		\item Exécuter le programme normalement (ce qui générera un fichier {\tt gmon.out} contenant les mesures réalisées)
		\item Lancer {\tt gprof}  avec les options {\tt -Q -b} pour n'afficher que l'essentiel.
	\end{enumerate}

\begin{block}{Exemple}
	\begin{verbatim}
	$ nvcc -pg matrix.cu
	$ ./a.out 
	$ gprof -Q -b a.out

		\item  Compiler le code
		\item Lancer {\tt nvprof} (en mode super utilisateur) : \\{\tt sudo /usr/local/cuda/bin/nvprof ./a.out }
	\end{enumerate}

\begin{block}{Exemple}

	$ nvcc matrix.cu
