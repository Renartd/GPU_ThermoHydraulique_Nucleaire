\begin{frame}[plain]
\frametitle{Debuggers : options de compilation}
	
	\begin{itemize}
		\item Il y a deux compilateurs utilisés :
		\begin{itemize}
			\item compilateur de l'hôte (gcc, clang, etc.) : code C/C++ de l'host
			\item NVCC : code du device
		\end{itemize}
		\medskip
		\item NVCC prend en charge certaines options du compilateur hôte :
		\begin{itemize}
			\item Si l'option n'est pas pris en charge par nvcc, utilisez {\tt -Xcompiler} pour le transmettre au compilateur de l'hôte, par exemple {\tt -Xcompiler -fopenmp} pour utiliser la bibliothèque openMP.
		\end{itemize}
		\medskip
		\item Options de débogage
		\begin{itemize}
			\item {\tt -g} : inclut les symboles de débogage de l'hôte
			\item {\tt -G} : inclut les symboles de débogage du device
			\item {\tt -lineinfo} : inclut les informations sur les lignes des symboles
		\end{itemize}
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[plain]
\frametitle{Debuggers : CUDA-MEMCHECK}
	
	\url{http://docs.nvidia.com/cuda/cuda-memcheck}\vspace{1em}
	
		\begin{itemize}
			\item Outil de debug pour la mémoire :
			\begin{itemize}
				\item Pas de recompilation nécessaire : {\tt cuda-memcheck ./exe}
			\end{itemize}
		\medskip
			\item Permet de détecter :
			\begin{itemize}
				\item Fuites de mémoire
				\item Erreurs de mémoire (OOM, accès mal aligné, instruction illégale, etc.)
				\item Accès concurrent
				\item Barrières mémoires illégales
				\item Mémoire non initialisée
			\end{itemize}
		\medskip
			\item Pour afficher les numéros de ligne, utilisez les options de compilation suivants : 
			\begin{itemize}
				\item Pas de recompilation nécessaire : {\tt -Xcompiler -rdynamic -lineinfo}
			\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}[plain]
\frametitle{Debuggers : CUDA-GDB}
	
	\url{http://docs.nvidia.com/cuda/cuda-gdb}\vspace{1em}
	
	\begin{itemize}
		\item - cuda-gdb est une extension de GDB :
		\begin{itemize}
			\item Permet un débogage du code CUDA et CPU
		\end{itemize}
		\medskip
		\item Fonctionne sous Linux et Macintosh
		\begin{itemize}
			\item Pour un débogueur Windows, utilisez NVIDIA Nsight Eclipse Edition ou Visual Studio Edition.
		\end{itemize}	
	\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}[plain]
\frametitle{Logiciel de profilage : NVPROF}
	
	\url{https://docs.nvidia.com/cuda/profiler-users-guide/\#nvprof}\vspace{1em}
	
	\begin{itemize}
		\item Logiciel de profilage en ligne de commande
		\begin{itemize}
			\item  Mesure le temps de calcul pour chaque kernel
			\item  Mesure le temps de transfert de la mémoire
			\item  Collecte les métriques et les événements
			\item  Prend en charge la hiérarchie complète des processus
			\item  Collecte les données pour NVIDIA Visual Profiler (NVVP)
			\item  Pas besoin de compiler
		\end{itemize}
	\end{itemize}
\end{frame}

		\end{itemize}
